116  void run_step(
117  int step,
118  IntegerRaster& infected,
119  IntegerRaster& susceptible,
120  IntegerRaster& total_populations,
121  IntegerRaster& dispersers,
122  std::vector<IntegerRaster>& exposed,
123  std::vector<IntegerRaster>& mortality_tracker,
124  IntegerRaster& died,
125  const std::vector<FloatRaster>& temperatures,
126  const FloatRaster& weather_coefficient,
127  Treatments<IntegerRaster, FloatRaster>& treatments,
128  IntegerRaster& resistant,

130  SpreadRate<IntegerRaster>& spread_rate, // out
131  QuarantineEscape<IntegerRaster>& quarantine, // out
132  const IntegerRaster& quarantine_areas,
133  const std::vector<std::vector<int>> movements,
134  const std::vector<std::vector<int>>& suitable_cells)
135  {
136  RadialDispersalKernel<IntegerRaster> natural_radial_kernel(
137  config_.ew_res,
138  config_.ns_res,
139  natural_kernel,
140  config_.natural_scale,
141  direction_from_string(config_.natural_direction),
142  config_.natural_kappa,
143  config_.shape);
144  RadialDispersalKernel<IntegerRaster> anthro_radial_kernel(
145  config_.ew_res,
146  config_.ns_res,
147  anthro_kernel,
148  config_.anthro_scale,
149  direction_from_string(config_.anthro_direction),
150  config_.anthro_kappa,
151  config_.shape);
152  DeterministicDispersalKernel<IntegerRaster> natural_deterministic_kernel(
153  natural_kernel,
154  dispersers,
155  config_.dispersal_percentage,
156  config_.ew_res,
157  config_.ns_res,
158  config_.natural_scale,
159  config_.shape);
160  DeterministicDispersalKernel<IntegerRaster> anthro_deterministic_kernel(
161  anthro_kernel,
162  dispersers,
163  config_.dispersal_percentage,
164  config_.ew_res,
165  config_.ns_res,
166  config_.anthro_scale,
167  config_.shape);
168  SwitchDispersalKernel<IntegerRaster> natural_selectable_kernel(
169  natural_kernel,
170  natural_radial_kernel,
171  natural_deterministic_kernel,
172  uniform_kernel,
173  natural_neighbor_kernel,
174  config_.deterministic);
175  SwitchDispersalKernel<IntegerRaster> anthro_selectable_kernel(
176  anthro_kernel,
177  anthro_radial_kernel,
178  anthro_deterministic_kernel,
179  uniform_kernel,
180  anthro_neighbor_kernel,
181  config_.deterministic);
182  DispersalKernel<IntegerRaster> dispersal_kernel(
183  natural_selectable_kernel,
184  anthro_selectable_kernel,
185  config_.use_anthropogenic_kernel,
186  config_.percent_natural_dispersal);
187  int mortality_simulation_year =
188  simulation_step_to_action_step(config_.mortality_schedule(), step);
189  // removal of dispersers due to lethal temperatures
190  if (config_.use_lethal_temperature && config_.lethal_schedule()[step]) {
191  int lethal_step =
192  simulation_step_to_action_step(config_.lethal_schedule(), step);
193  simulation_.remove(
194  infected,
195  susceptible,
196  temperatures[lethal_step],
197  config_.lethal_temperature,
198  suitable_cells);
199  }
200  // actual spread
201  if (config_.spread_schedule()[step]) {
202  simulation_.generate(
203  dispersers,
204  infected,
205  config_.weather,
206  weather_coefficient,
207  config_.reproductive_rate,
208  suitable_cells);
209 
210  simulation_.disperse_and_infect(
211  step,
212  dispersers,
213  susceptible,
214  exposed,
215  infected,
216  mortality_tracker[mortality_simulation_year],
217  total_populations,
218  outside_dispersers,
219  config_.weather,
220  weather_coefficient,
221  dispersal_kernel,
222  suitable_cells,
223  config_.establishment_probability);
224  if (config_.use_overpopulation_movements) {
225  simulation_.move_overpopulated_pests(
226  susceptible,
227  infected,
228  total_populations,
229  outside_dispersers,
230  anthro_selectable_kernel,
231  suitable_cells,
232  config_.overpopulation_percentage,
233  config_.leaving_percentage);
234  }
235  if (config_.use_movements) {
236  last_index = simulation_.movement(
237  infected,
238  susceptible,
239  mortality_tracker[mortality_simulation_year],
240  total_populations,
241  step,
242  last_index,
243  movements,
244  config_.movement_schedule);
245  }
246  }
247  // treatments
248  if (config_.use_treatments) {
249  bool managed = treatments.manage(
250  step, infected, exposed, susceptible, resistant, suitable_cells);
251  if (managed && config_.use_mortality) {
252  // same conditions as the mortality code below
253  // TODO: make the mortality timing available as a separate function in
254  // the library or simply go over all valid cohorts
255  if (mortality_simulation_year >= config_.first_mortality_year - 1) {
256  auto max_index =
257  mortality_simulation_year - (config_.first_mortality_year - 1);
258  for (int age = 0; age <= max_index; age++) {
259  treatments.manage_mortality(
260  step, mortality_tracker[age], suitable_cells);
261  }
262  }
263  }
264  }
265  if (config_.use_mortality && config_.mortality_schedule()[step]) {
266  // only run to the current year of simulation
267  // (first year is 0):
268  // max index == sim year
269  // reduced by first time when trees start dying
270  // (counted from 1: first year == 1)
271  // e.g. for sim year 3, year dying 4, max index is 0
272  // max index = sim year - (dying year - 1)
273  // index is negative before we reach the year
274  // (so we can skip these years)
275  // sim year - (dying year - 1) < 0
276  // sim year < dying year - 1
277  // TODO: died.zero(); should be done by the caller if needed, document!
278  simulation_.mortality(
279  infected,
280  config_.mortality_rate,
281  mortality_simulation_year,
282  config_.first_mortality_year - 1,
283  died,
284  mortality_tracker,
285  suitable_cells);
286  }
287  // compute spread rate
288  if (config_.use_spreadrates && config_.spread_rate_schedule()[step]) {
289  unsigned rates_step =
290  simulation_step_to_action_step(config_.spread_rate_schedule(), step);
291  spread_rate.compute_step_spread_rate(infected, rates_step, suitable_cells);
292  }
293  // compute quarantine escape
294  if (config_.use_quarantine && config_.quarantine_schedule()[step]) {
295  unsigned action_step =
296  simulation_step_to_action_step(config_.quarantine_schedule(), step);
297  quarantine.infection_escape_quarantine(
298  infected, quarantine_areas, action_step, suitable_cells);
299  }
300  }
301 };
