193  void mortality(
194  IntegerRaster& infected,
195  double mortality_rate,
196  int current_year,
197  int first_mortality_year,
198  IntegerRaster& mortality,
199  std::vector<IntegerRaster>& mortality_tracker_vector,
200  const std::vector<std::vector<int>>& suitable_cells)
201  {
202  if (current_year >= (first_mortality_year)) {
203  int mortality_current_year = 0;
204  int max_year_index = current_year - first_mortality_year;
205 
206  for (auto indices : suitable_cells) {
207  int i = indices[0];
208  int j = indices[1];
209  for (int year_index = 0; year_index <= max_year_index; year_index++) {
210  int mortality_in_year_index = 0;
211  if (mortality_tracker_vector[year_index](i, j) > 0) {
212  mortality_in_year_index =
213  mortality_rate * mortality_tracker_vector[year_index](i, j);
214  mortality_tracker_vector[year_index](i, j) -=
215  mortality_in_year_index;
216  mortality(i, j) += mortality_in_year_index;
217  mortality_current_year += mortality_in_year_index;
218  if (infected(i, j) > 0) {
219  infected(i, j) -= mortality_in_year_index;
220  }
221  }
222  }
223  }

246  unsigned movement(
247  IntegerRaster& infected,
248  IntegerRaster& susceptible,
249  IntegerRaster& mortality_tracker,
250  IntegerRaster& total_hosts,
251  unsigned step,
252  unsigned last_index,
253  const std::vector<std::vector<int>>& movements,
254  std::vector<unsigned> movement_schedule)
255  {
256  UNUSED(mortality_tracker); // Mortality is not supported by movements.
257  for (unsigned i = last_index; i < movements.size(); i++) {
258  auto moved = movements[i];
259  unsigned move_schedule = movement_schedule[i];
260  if (move_schedule != step) {
261  return i;
262  }
263  int infected_moved = 0;
264  int susceptible_moved = 0;
265  int total_hosts_moved = 0;
266  double inf_ratio = 0;
267  int row_from = moved[0];
268  int col_from = moved[1];
269  int row_to = moved[2];
270  int col_to = moved[3];
271  int hosts = moved[4];
272  if (hosts > total_hosts(row_from, col_from)) {
273  total_hosts_moved = total_hosts(row_from, col_from);
274  }
275  else {
276  total_hosts_moved = hosts;
277  }
278  if (infected(row_from, col_from) > 0
279  && susceptible(row_from, col_from) > 0) {
280  inf_ratio = double(infected(row_from, col_from))
281  / double(total_hosts(row_from, col_from));
282  int infected_mean = total_hosts_moved * inf_ratio;
283  if (infected_mean > 0) {
284  if (movement_stochasticity_) {
285  std::poisson_distribution<int> distribution(infected_mean);
286  infected_moved = distribution(generator_);
287  }
288  else {
289  infected_moved = infected_mean;
290  }
291  }
292  if (infected_moved > infected(row_from, col_from)) {
293  infected_moved = infected(row_from, col_from);
294  }
295  if (infected_moved > total_hosts_moved) {
296  infected_moved = total_hosts_moved;
297  }
298  susceptible_moved = total_hosts_moved - infected_moved;
299  if (susceptible_moved > susceptible(row_from, col_from)) {
300  susceptible_moved = susceptible(row_from, col_from);
301  }
302  }
303  else if (
304  infected(row_from, col_from) > 0
305  && susceptible(row_from, col_from) == 0) {
306  infected_moved = total_hosts_moved;
307  }
308  else if (
309  infected(row_from, col_from) == 0
310  && susceptible(row_from, col_from) > 0) {
311  susceptible_moved = total_hosts_moved;
312  }
313  else {
314  continue;
315  }
316 
317  infected(row_from, col_from) -= infected_moved;
318  susceptible(row_from, col_from) -= susceptible_moved;
319  total_hosts(row_from, col_from) -= total_hosts_moved;
320  infected(row_to, col_to) += infected_moved;
321  susceptible(row_to, col_to) += susceptible_moved;
322  total_hosts(row_to, col_to) += total_hosts_moved;
323  }
324  return movements.size();
325  }

336  void generate(
337  IntegerRaster& dispersers,
338  const IntegerRaster& infected,
339  bool weather,
340  const FloatRaster& weather_coefficient,
341  double reproductive_rate,
342  const std::vector<std::vector<int>>& suitable_cells)
343  {
344  double lambda = reproductive_rate;
345  for (auto indices : suitable_cells) {
346  int i = indices[0];
347  int j = indices[1];
348  if (infected(i, j) > 0) {
349  if (weather)
350  lambda = reproductive_rate * weather_coefficient(i, j);
351  int dispersers_from_cell = 0;
352  if (dispersers_stochasticity_) {
353  std::poisson_distribution<int> distribution(lambda);
354  for (int k = 0; k < infected(i, j); k++) {
355  dispersers_from_cell += distribution(generator_);
356  }
357  }
358  else {
359  dispersers_from_cell = lambda * infected(i, j);
360  }
361  dispersers(i, j) = dispersers_from_cell;
362  }
363  else {
364  dispersers(i, j) = 0;
365  }
366  }
367  }

412  template<typename DispersalKernel>
413  void disperse(
414  const IntegerRaster& dispersers,
415  IntegerRaster& susceptible,
416  IntegerRaster& exposed_or_infected,
417  IntegerRaster& mortality_tracker,
418  const IntegerRaster& total_populations,
419  std::vector<std::tuple<int, int>>& outside_dispersers,
420  bool weather,
421  const FloatRaster& weather_coefficient,
422  DispersalKernel& dispersal_kernel,
423  const std::vector<std::vector<int>>& suitable_cells,
424  double establishment_probability = 0.5)
425  {
426  std::uniform_real_distribution<double> distribution_uniform(0.0, 1.0);
427  int row;
428  int col;
429 
430  for (auto indices : suitable_cells) {
431  int i = indices[0];
432  int j = indices[1];
433  if (dispersers(i, j) > 0) {
434  for (int k = 0; k < dispersers(i, j); k++) {
435  std::tie(row, col) = dispersal_kernel(generator_, i, j);
436  if (row < 0 || row >= rows_ || col < 0 || col >= cols_) {
437  // export dispersers dispersed outside of modeled area
438  outside_dispersers.emplace_back(std::make_tuple(row, col));
439  continue;
440  }
441  if (susceptible(row, col) > 0) {
442  double probability_of_establishment =
443  (double)(susceptible(row, col))
444  / total_populations(row, col);
445  double establishment_tester = 1 - establishment_probability;
446  if (establishment_stochasticity_)
447  establishment_tester = distribution_uniform(generator_);
448 
449  if (weather)
450  probability_of_establishment *= weather_coefficient(i, j);
451  if (establishment_tester < probability_of_establishment) {
452  exposed_or_infected(row, col) += 1;
453  susceptible(row, col) -= 1;
454  if (model_type_ == ModelType::SusceptibleInfected) {
455  mortality_tracker(row, col) += 1;
456  }
457  else if (
458  model_type_ == ModelType::SusceptibleExposedInfected) {
459  // no-op
460  }
461  else {
462  throw std::runtime_error(
463  "Unknown ModelType value in "
464  "Simulation::disperse()");
465  }
466  }
467  }
468  }
469  }

502  template<typename DispersalKernel>
503  void move_overpopulated_pests(
504  IntegerRaster& susceptible,
505  IntegerRaster& infected,
506  const IntegerRaster& total_hosts,
507  std::vector<std::tuple<int, int>>& outside_dispersers,
508  DispersalKernel& dispersal_kernel,
509  const std::vector<std::vector<int>>& suitable_cells,
510  double overpopulation_percentage,
511  double leaving_percentage)
512  {
513  struct Move
514  {
515  RasterIndex row;
516  RasterIndex col;
517  int count;
518  };
519  std::vector<Move> moves;
520 
521  // Identify the moves. Remove from source cells.
522  for (auto indices : suitable_cells) {
523  int i = indices[0];
524  int j = indices[1];
525  int original_count = infected(i, j);
526  // No move with only one infected host (one unit).
527  if (original_count <= 1)
528  continue;
529  // r = I / (I + S)
530  // r = I / (I + S + E_1 + E_2 + ...)
531  double ratio = original_count / double(total_hosts(i, j));
532  if (ratio >= overpopulation_percentage) {
533  int row;
534  int col;
535  std::tie(row, col) = dispersal_kernel(generator_, i, j);
536  // for leaving_percentage == 0.5
537  // 2 infected -> 1 leaving
538  // 3 infected -> 1 leaving
539  int leaving = original_count * leaving_percentage;
540  susceptible(i, j) += leaving;
541  infected(i, j) -= leaving;
542  if (row < 0 || row >= rows_ || col < 0 || col >= cols_) {
543  // Collect pests dispersed outside of modeled area.
544  outside_dispersers.reserve(outside_dispersers.size() + leaving);
545  for (int pest = 0; pest < leaving; ++pest)
546  outside_dispersers.emplace_back(row, col);
547  continue;
548  }
549  // Doing the move here would create inconsistent results as some
550  // target cells would be evaluated after the moved pest arrived,
551  // possibly triggering another move.
552  // So, instead, we just collect them and apply later. (The pest is in
553  // the air when in the moves vector.)
554  moves.push_back(Move({row, col, leaving}));
555  }
556  }
557  // Perform the moves to target cells.
558  for (const auto& move : moves) {
559  RasterIndex row = move.row;
560  RasterIndex col = move.col;
561  int leaving = move.count;
562  // The target cell can accept all.
563  if (susceptible(row, col) >= leaving) {
564  susceptible(row, col) -= leaving;
565  infected(row, col) += leaving;
566  }
567  // More pests than the target cell can accept.
568  // This can happen if there is simply not enough S hosts to accomodate all
569  // the pests moving from the source or if multiple sources end up in the
570  // same target cell and there is not enough S hosts to accomodate all of
571  // them. The pests just disappear in both cases.
572  else {
573  leaving = susceptible(row, col);
574  susceptible(row, col) -= leaving;
575  infected(row, col) += leaving;
576  }
577  }
578  }

621  void infect_exposed(
622  unsigned step,
623  std::vector<IntegerRaster>& exposed,
624  IntegerRaster& infected,
625  IntegerRaster& mortality_tracker)
626  {
627  if (model_type_ == ModelType::SusceptibleExposedInfected) {
628  if (step >= latency_period_) {
629  // Oldest item needs to be in the front
630  auto& oldest = exposed.front();
631  // Move hosts to infected raster
632  infected += oldest;
633  mortality_tracker += oldest;
634  // Reset the raster
635  // (hosts moved from the raster)
636  oldest.fill(0);
637  }
638  // Age the items and the used one to the back
639  // elements go one position to the left
640  // new oldest goes to the front
641  // old oldest goes to the back
642  rotate_left_by_one(exposed);
643  }
644  else if (model_type_ == ModelType::SusceptibleInfected) {
645  // no-op
646  }
647  else {
648  throw std::runtime_error(
649  "Unknown ModelType value in Simulation::infect_exposed()");
650  }
651  }

679  template<typename DispersalKernel>
680  void disperse_and_infect(
